Структура проекта GST-CMDB и назначение файлов/папок

config.php               — настройки подключения к базе, параметры CMDB
includes/db_connect.php  — подключение к базе через PDO
includes/csrf.php        — генерация и проверка CSRF-токенов для форм

README.md                — подробная инструкция по установке, настройкам, безопасности, импорту/экспорту
structure.txt            — этот файл с комментариями к каждому разделу/файлу

dashboard.php            — главная (для авторизованных): метрики, цветовые блоки (дашборд)
index.php                — стартовая страница/редирект на dashboard, гостевое приветствие

auth/                    — авторизация и логика сессий
  login.php              — вход пользователя
  logout.php             — выход
  register_admin.php     — создание первого администратора
  check_auth.php         — include для контроля ролей и наличия входа

assets/                  — оборудование
  index.php              — список; импорт/экспорт csv, фильтр, удаление, badge-статус
  add.php                — добавить (CRUD, CSRF, журналирование)
  edit.php               — редактирование
  delete.php             — удаление с подтверждением
  software.php           — управление связями с ПО
  export_csv.php         — экспорт оборудования в CSV
  import.php             — массовый импорт оборудования из CSV
  sample.csv             — пример csv для импорта

software/                — программное обеспечение
  index.php              — список, фильтр, тип-бейдж, экспорт/импорт csv, удаление
  add.php                — добавление
  edit.php               — редактирование
  delete.php             — удаление
  equipment.php          — список устройств, где установлено ПО
  export_csv.php         — экспорт ПО в CSV
  import.php             — импорт ПО из CSV
  sample.csv             — пример csv для импорта

licenses/                — лицензии
  index.php              — список, экспорт, импорт, поиск, цветные статусы, удаление
  add.php                — добавление
  edit.php               — редактирование
  delete.php             — удаление
  export_csv.php         — экспорт лицензий в CSV
  import.php             — массовый импорт лицензий из CSV
  sample.csv             — пример csv для импорта

settings/                — настройка email, периода уведомлений (только admin)
  index.php              — форма для правки параметров, CSRF, логи

logs/                    — журнал действий
  index.php              — просмотр последних действий (доступ только admin)

templates/               — шаблоны (header/footer)

cron_notify.php          — скрипт для ежедневных e-mail уведомлений (исп. параметры из settings)
db_schema.sql            — SQL-дамп для инициализации таблиц

---
import.php во всех разделах реализует массовую загрузку записей из CSV-файлов
sample.csv — шаблон для обработки, структура соответствует заголовкам в импорте
функция доступна только для admin/operator, выдаёт отчёт по результату
